let privacyOn=false;

// "Anna Müller" → "A. M."
function maskName(name){
  if(!name||!privacyOn)return name;
  return name.trim().split(/\s+/).map(w=>w[0].toUpperCase()+'.').join(' ');
}
// "anna@beispiel.de" → "a***@***.de"
function maskEmail(email){
  if(!email||!privacyOn)return email;
  const[local,domain]=email.split('@');
  if(!domain)return email[0]+'***';
  const ext=domain.lastIndexOf('.');
  const domMasked=ext>0?'***'+domain.slice(ext):'***';
  return(local[0]||'')+'***@'+domMasked;
}
// "+49 171 1234567" → "+49 ***"
function maskPhone(phone){
  if(!phone||!privacyOn)return phone;
  const m=phone.match(/^(\+?\d{1,4}[\s-]?)/);
  return(m?m[1]:'')+'***';
}
// "Musterstr. 1, 10115 Berlin" → "***, Berlin"
function maskAddr(addr){
  if(!addr||!privacyOn)return addr;
  const parts=addr.split(',');
  if(parts.length<2)return'***';
  return'***, '+parts.slice(-1)[0].trim();
}
// Wrap in styled tag when privacy active
function pTag(text){
  return privacyOn?`<em class="priv-tag">${text}</em>`:text;
}

// Fügt Platzhalter an der aktuellen Cursorposition ein.
// activeId = zuletzt fokussiertes Feld; fallbackId = Fallback wenn kein Fokus
function insertPlaceholder(fallbackId, placeholder, altFallbackId){
  // Welches Feld war zuletzt aktiv?
  const focused = document.activeElement;
  let target = null;
  const candidates = [fallbackId, altFallbackId].filter(Boolean);
  if(focused && (focused.tagName==='TEXTAREA'||focused.tagName==='INPUT') && candidates.includes(focused.id)){
    target = focused;
  } else {
    // Kein passendes Feld fokussiert → Standard-Fallback
    target = document.getElementById(altFallbackId||fallbackId);
  }
  if(!target) return;
  const start = target.selectionStart ?? target.value.length;
  const end   = target.selectionEnd   ?? target.value.length;
  target.value = target.value.slice(0,start) + placeholder + target.value.slice(end);
  const pos = start + placeholder.length;
  target.focus();
  target.setSelectionRange(pos, pos);
}

const MONTHS_DE = ['Januar','Februar','März','April','Mai','Juni','Juli','August','September','Oktober','November','Dezember'];

function renderCalendar() {
  document.getElementById('cal-month-label').textContent = MONTHS_DE[calMonth] + ' ' + calYear;
  const today = new Date(); today.setHours(0,0,0,0);
  const firstDay = new Date(calYear, calMonth, 1);
  // Monday = 0 offset
  let startOffset = firstDay.getDay() - 1;
  if(startOffset < 0) startOffset = 6;
  const daysInMonth = new Date(calYear, calMonth+1, 0).getDate();
  const daysInPrev = new Date(calYear, calMonth, 0).getDate();

  let html = '';
  let dayCount = 0;
  const totalCells = Math.ceil((startOffset + daysInMonth) / 7) * 7;

  for(let i = 0; i < totalCells; i++) {
    let day, month, year, isOther = false;
    if(i < startOffset) {
      day = daysInPrev - startOffset + i + 1;
      month = calMonth - 1; year = calYear;
      if(month < 0) { month = 11; year--; }
      isOther = true;
    } else if(i - startOffset >= daysInMonth) {
      day = i - startOffset - daysInMonth + 1;
      month = calMonth + 1; year = calYear;
      if(month > 11) { month = 0; year++; }
      isOther = true;
    } else {
      day = i - startOffset + 1;
      month = calMonth; year = calYear;
    }
    const dateStr = `${year}-${String(month+1).padStart(2,'0')}-${String(day).padStart(2,'0')}`;
    const cellDate = new Date(year, month, day);
    const isToday = cellDate.getTime() === today.getTime();
    const dayShootings = (DB.shootings||[]).filter(s => s.date === dateStr);
    const hasShooting = dayShootings.length > 0;
    const dots = dayShootings.slice(0,2).map(s => {
      const dotColor = s.status==='confirmed' ? 'var(--success)' : s.status==='cancelled' ? 'var(--danger)' : 'var(--warn)';
      return '<div style="color:'+dotColor+';font-size:9px;overflow:hidden;text-overflow:ellipsis;white-space:nowrap">'+(s.title||'Shooting')+'</div>';
    }).join('');
    html += `<div class="cal-day${isOther?' other-month':''}${isToday?' today':''}${hasShooting?' has-shooting':''}" onclick="calDayClick('${dateStr}')">
      <div class="cal-dn">${day}</div>
      <div class="cal-dot">${dots}</div>
    </div>`;
  }
  document.getElementById('cal-body').innerHTML = html;
}

    // Default 2h duration
    const endH = (parseInt(h) + 2) % 24;
    dtEnd = dateClean + 'T' + pad(endH) + pad(m) + '00';
  } else {
    dtStart = dateClean;
    dtEnd   = dateClean;
  }

  const now = new Date().toISOString().replace(/[-:.]/g,'').slice(0,15) + 'Z';
  const uid_val = s.id + '@tfp-manager';

  // Build reminder block
  let alarmBlock = '';
  if(s.reminder) {
    alarmBlock = [
      'BEGIN:VALARM',
      'TRIGGER:-P' + s.reminder + 'D',
      'ACTION:DISPLAY',
      'DESCRIPTION:Erinnerung: ' + (s.title||'Shooting'),
      'END:VALARM'
    ].join('\r\n');
  }

  const lines = [
    'BEGIN:VCALENDAR',
    'VERSION:2.0',
    'PRODID:-//TfP Manager//DE',
    'CALSCALE:GREGORIAN',
    'METHOD:PUBLISH',
    'BEGIN:VEVENT',
    'UID:' + uid_val,
    'DTSTAMP:' + now,
    s.time ? 'DTSTART:' + dtStart : 'DTSTART;VALUE=DATE:' + dtStart,
    s.time ? 'DTEND:' + dtEnd   : 'DTEND;VALUE=DATE:' + dtEnd,
    'SUMMARY:📷 TfP Shooting – ' + (s.title||'Shooting'),
    s.location ? 'LOCATION:' + s.location : '',
    'DESCRIPTION:' + [
      'Fotograf/in: ' + (p.name||''),
      s.notes ? 'Notizen: ' + s.notes : '',
      'Status: ' + (STATUS_LABEL[s.status]||'Offen'),
      '',
      'Erstellt mit TfP Manager'
    ].filter(Boolean).join('\\n'),
    'STATUS:' + (s.status==='confirmed' ? 'CONFIRMED' : s.status==='cancelled' ? 'CANCELLED' : 'TENTATIVE'),
    alarmBlock,
    'END:VEVENT',
    'END:VCALENDAR'
  ].filter(l => l !== '').join('\r\n');

  const blob = new Blob([lines], {type:'text/calendar;charset=utf-8'});
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  const safeName = (s.title||'Shooting').replace(/[^a-z0-9]/gi,'-').toLowerCase();
  a.href = url;
  a.download = 'tfp-' + safeName + '-' + s.date + '.ics';
  a.click();
  setTimeout(() => URL.revokeObjectURL(url), 5000);
  showToast('✓ Kalenderdatei (.ics) gespeichert');
}

function checkReminders() {
  if(!DB.shootings) return;
  const today = new Date(); today.setHours(0,0,0,0);
  DB.shootings.forEach(s => {
    if(!s.reminder || s.status==='cancelled') return;
    const d = new Date(s.date); d.setHours(0,0,0,0);
    const daysUntil = Math.round((d - today) / 86400000);
    if(daysUntil === parseInt(s.reminder) && !s._reminded) {
      showToast(`⏰ Erinnerung: "${s.title}" in ${daysUntil} Tag(en)`, 6000);
      s._reminded = true;
    }
  });
}

function importJSON(event) {
  const file = event.target.files[0];
  if(!file) return;
  const reader = new FileReader();
  reader.onload = e => {
    try {
      const parsed = JSON.parse(e.target.result);
      if(!parsed.photographer || !Array.isArray(parsed.contracts)) {
        alert('❌ Ungültige Backup-Datei. Bitte nur Dateien laden die mit dieser App exportiert wurden.');
        return;
      }
      if(!confirm('Backup laden?\n\n⚠ Alle aktuellen Daten werden überschrieben!\n\nDatei: ' + file.name)) return;
      DB = parsed;
      [{id:'t1',name:'Standard TfP-Vertrag',body:TPL1},{id:'t2',name:'Erweiterter TfP-Vertrag (inkl. MUA)',body:TPL2},{id:'t3',name:'TfP kommerziell – OnlyFans',body:TPL3},{id:'t4',name:'TfP kommerziell – Patreon',body:TPL4}].forEach(t=>{
        if(!DB.templates) DB.templates=[];
        if(!DB.templates.find(x=>x.id===t.id)) DB.templates.push(t);
      });
      if(!DB.emailTemplates||!DB.emailTemplates.length) DB.emailTemplates=defaultDB().emailTemplates;
      if(!DB.stats) DB.stats=[];
      if(!DB.shootings) DB.shootings=[];
      persist();
      document.getElementById('sb-info').textContent = DB.photographer.name ? '👤 '+DB.photographer.name : '⚠ Profil ausfüllen';
      go('dashboard');
      showToast('✓ Backup erfolgreich geladen');
    } catch(err) {
      alert('❌ Fehler beim Lesen der Datei: ' + err.message);
    }
  };
  reader.readAsText(file);
}

// ── TOAST ────────────────────────────────────────────────────────────────────
function showToast(msg, duration=3000) {
  let t = document.getElementById('app-toast');
  if(!t) {
    t = document.createElement('div');
    t.id = 'app-toast';
    t.style.cssText = 'position:fixed;bottom:28px;left:50%;transform:translateX(-50%) translateY(16px);background:var(--card);border:1px solid var(--accent);color:var(--accent);padding:10px 22px;border-radius:var(--rs);font-size:13px;font-family:"Jost",sans-serif;font-weight:500;z-index:9999;opacity:0;transition:opacity .22s,transform .22s;pointer-events:none;white-space:nowrap;box-shadow:0 4px 20px rgba(0,0,0,.4)';
    document.body.appendChild(t);
  }
  t.textContent = msg;
  t.style.opacity = '1';
  t.style.transform = 'translateX(-50%) translateY(0)';
  clearTimeout(t._timer);
  t._timer = setTimeout(() => {
    t.style.opacity = '0';
    t.style.transform = 'translateX(-50%) translateY(16px)';
  }, duration);
}
